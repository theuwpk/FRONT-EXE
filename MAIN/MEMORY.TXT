$zipUrl = "https://example.com/arquivo.zip"
$exeName = "programa.exe"
$tempFolder = [System.IO.Path]::GetTempPath()
$zipFile = Join-Path $tempFolder "arquivo.zip"
$extractFolder = Join-Path $tempFolder "extraido"
$exePath = Join-Path $extractFolder $exeName

if (-Not (Test-Path $zipFile)) { Invoke-WebRequest -Uri $zipUrl -OutFile $zipFile }
if (-Not (Test-Path $extractFolder)) { Expand-Archive -Path $zipFile -DestinationPath $extractFolder -Force }
if (Test-Path $exePath) { Start-Process -FilePath $exePath }
